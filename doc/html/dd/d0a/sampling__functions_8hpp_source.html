73  * use case envisioned is for sampling from multilocus populations,

78  */

84

86  const double pos) {
87  return pop.mutations[p.first].pos < pos;
88  };
89

91  const std::pair<std::size_t, uint_t> &p) {
92  return pos < pop.mutations[p.first].pos;
