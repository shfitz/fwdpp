12 /*!
13  \file gamete_cleaner.hpp
14 
15  This file handles the "pruning" of fixations from gametes.
16 
17  Whether or not any pruning happens depends on policies.
18 
19  Version 0.4.7 of the library made a major change resulting in improved
20  run-time performance, especially for low mutation rates.
21 
22  Consider the case of neutral mutations occurring at rate mu (per site,
23  per generation). The rate of fixation is also mu. Thus, when mu is low,
24  the rate of fixation is low, and the expected time in generations between
25  fixations is 1/mu. In other words, calling gamete_cleaner every generation
26  is an unnecessary run-time cost. Now, we check if any fixations exist, and
27  return
28  if there are none. We check separately for neutral- and non-neutral
29  fixations.
30 
31  Naively, we could simply scan the mutations container of the entire
32  simulation for fixations.
33  But, we can do better than that because:
34 
35  1. Fixations are, by definition, found in all gametes. Thus, it suffices to
36  check the
37  first (extant) gamete.
38 
39  2. When we find the first fixation in a gamete, it is by definition the
40  fixation with the smallest
41  position, out of all fixations. We can store its value and use it to look up
42  the first fixation
43  in all of the remaining gametes. Searching for the first fixation in this
44  way avoids cache misses that
45  are unavoidable when we do out-of-order lookups in "mcounts" for the
46  remaining fixations.
47 */

220  const idiom_wrapper &iw) noexcept
221  /*!
222  The two overloads of gamete_cleaner dispatch the above policies into
223  this function.
224  */
225  {
226  auto extant_gamete
227  = next_extant_gamete(gametes.begin(), gametes.end());
228  const auto fixation_n = ff(extant_gamete->mutations);
229  bool neutral_fixations_exist
230  = (fixation_n != extant_gamete->mutations.cend());
231  const auto fixation_s = ff(extant_gamete->smutations);
232  bool selected_fixations_exist

234  if (!neutral_fixations_exist && !selected_fixations_exist)
235  return;

239  const auto fixation_n_value

241  ? typename decltype(fixation_n)::value_type()
242  : *fixation_n;
243  const auto fixation_s_value
244  = (fixation_s == extant_gamete->smutations.cend())
245  ? typename decltype(fixation_s)::value_type()
246  : *fixation_s;
247  while (extant_gamete < gend)
248  {
249  if (neutral_fixations_exist)
250  {
251  iw(extant_gamete->mutations, fixation_n_value);
252  }
253  if (selected_fixations_exist)
254  {
255  iw(extant_gamete->smutations, fixation_s_value);

262  template <typename gcont_t, typename fixation_finder>
263  std::pair<bool, bool>
264  fixation_finder_search_all(const gcont_t &gametes,
265  const fixation_finder &ff) noexcept
266  /*!
267  Determines whether a fixation exists in ANY gamete.
268  */
269  {
270  bool neutral = false, selected = false;
271  for (const auto &g : gametes)
272  {
273  if (!neutral)
274  {
275  auto itr = ff(g.mutations);
276  if (itr != g.mutations.cend())
277  {
278  neutral = true;
279  }
280  }
281  if (!selected)
282  {
283  auto itr = ff(g.smutations);
284  if (itr != g.smutations.cend())
285  {
286  selected = true;
287  }
288  }
289  if (neutral && selected)
290  return std::make_pair(neutral, selected);
291  }
292  return std::make_pair(neutral, selected);
293  }
