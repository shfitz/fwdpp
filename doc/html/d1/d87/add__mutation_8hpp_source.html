29  /*!
30  \brief Helper function for implementing fwdpp::add_mutation
31 
32  If \a new_mutation already exists in \a mutations, its location is
33  returned.
34 
35  Otherwise, this function puts the new mutation into the mutations
36  container and
37  updates other objects as needed.
38 
39  \return A key telling where \a new_mutation exists within \a
40  mutations.
41 
42  \note This function requires that operator== be defined for the
43  mutation type.
44 
45  \note In the event that \a new_mutation is added to \a mutations, the
46  count for that
47  mutations will be zero.
48 
49  \ingroup sugar
50  */
51  {
52  // If the mutation already exists, let's not create it...
53  auto mitr = std::find(std::begin(mutations), std::end(mutations),
54  new_mutation);
55  if (mitr != mutations.end())
56  return std::size_t(std::distance(std::begin(mutations), mitr));
57 
58  /*
59  If we are here, then new_mutation does not currently exist, so we
60  have to add it.
61 
62  We do this using fwdpp's recycling method. But, we don't need to
63  create recycling bin,
64  as we're just looking to add one mutation. Instead, we can
65  simply search for an extinct
66  mutation.
67  */
68  // find an extinct mutation, if one exists
69  auto extinct_mut
70  = std::find(std::begin(mcounts), std::end(mcounts), 0);
71  std::size_t mindex = mcounts.size(); // this will be the correct
72  // value if we use the else
73  // block below
74  if (extinct_mut != std::end(mcounts)) // then we can recycle
75  {
76  auto dist
77  = std::distance(std::begin(mcounts), extinct_mut);
78  mutations[dist] = std::move(
79  new_mutation); // move new mutation into place
80  mindex = std::size_t(dist); // update our value
81  }
82  else
83  {
84  // cannot recycle, so add it to end
85  mutations.emplace_back(std::move(new_mutation));
86  mcounts.push_back(0); // Add a place for this variant
87  }
88  return mindex;
89  }
