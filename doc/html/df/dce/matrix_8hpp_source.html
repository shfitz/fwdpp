21  struct data_matrix
22  /*!
23  * \brief Genotype or haplotype matrix.
24  *
25  * This type uses std::vector<std::int8_t> to hold a matrix
26  * representing the genotypes for a set of diploids.
27  *
28  * For a haplotype matrix of n individuals, the data represent
29  * 2n rows with a 0/1 encoding representing ancestral/derived.
30  *
31  * For a genotype matrix of n individuals, the data represent
32  * n rows with a 0/1/2 encoding for the number of copies of the
33  * derived mutation.
34  *
35  * The data layout is row-major (aka "C-style") ordering,
36  * facilitating compatibility with GSL matrix types,
37  * NumPy 2D arrays, etc. Note that GSL matrices may be
38  * constructed using gsl_matrix_view_array or
39  * gsl_matrix_const_view_array for cases where a matrix of
40  * only neutral or only selected mutations is needed.
41  *
42  * We use the 8-bit integer type to save space. In practice,
43  * one may convert (via copy) to other types for operations like
44  * regression.
45  *
46  * \note This type is not constructed directly, but rather returned
47  * by other functions.
48  */
49  {
50  //! Data for neutral mutations.
51  std::vector<std::int8_t> neutral;
52  //! Data for selected mutations.
53  std::vector<std::int8_t> selected;
54  //! Positions of neutral mutations. Same order as matrix column order
55  std::vector<double> neutral_positions;
56  //! Positions of selected mutations. Same order as matrix column order
57  std::vector<double> selected_positions;
58  //! Frequencies of neutral mutations in entire population. Same order
59  //! as matrix column order
60  std::vector<double> neutral_popfreq;
61  //! Frequencies of selected mutations in entire population. Same order
62  //! as matrix column order
63  std::vector<double> selected_popfreq;
64  //! Number of rows in the matrix
65  std::size_t nrow;

73  * type can be easily stack-allocated. Intended use cases are
74  * Cython,

101  /*!
102  * For a sample defined by a set of diploids, obtain the keys corresponding
103  * to all mutations in that sample.
104  *
105  * \param pop The population object
106  * \param individuals The indexes of individuals in the sample
107  * \param include_neutral If true, obtain keys for neutral sites
108  * \param include_selected If true, obtain keys for selected sites
109  *
110  * \return A pair of vectors representing neutral and selected keys,
111  * respectively. The value
112  * type of each vector is std::pair<std::size_t,fwdpp::uint_t>. The first
113  * value is the mutation
114  * key, and the second is its frequency in the sample.
115  *
116  * Several comments are required:
117  *
118  * 1. Return values are unsorted with respect to anything meaningful. If
119  * you wish to sort on position,
120  * etc., do so yourself.
121  * 2. In general, any manipulation of the keys is possible. For example,
122  * removing mutations that are fixed
123  * in the sample, or whose frequency is or is not within some desired
124  * range, is easily doable via the
125  * "erase/remove" idiom.
126  * 3. Keys from multiple samples can be merged to form new vectors where
127  * the key elements are unique and
128  * the frequencies are summed, all using standard C++.
129  */
130  {
131  return data_matrix_details::mutation_keys(
132  pop.diploids, individuals, pop.gametes, pop.mcounts,
133  include_neutral, include_selected,
134  typename poptype::popmodel_t());
135  }
