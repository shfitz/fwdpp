328  template <typename genetic_param_holder,
329  typename mutation_handling_policy, typename poptype>
330  std::pair<mut_rec_intermediates, mut_rec_intermediates>
331  generate_offspring(const gsl_rng* r,
332  const std::pair<std::size_t, std::size_t> parents,
333  const mutation_handling_policy& mutation_policy,
334  poptype& pop, genetic_param_holder& genetics,
335  typename poptype::diploid_t& offspring)
336  /// \brief Generate offspring gametes and return breakpoints plus mutation keys
337  ///
338  /// \param r Random number generator
339  /// \param parents Indexes of the offspring parents in \a pop
340  /// \param mutation_policy Either all_mutations or selected_variants_only. See below.
341  /// \param pop Either fwdpp::poptypes::slocuspop or fwdpp::poptypes::mlocuspop.
342  /// \param genetics A duck type of fwdpp::genetic_parameters.
343  /// \param offspring The offspring for which we will generate gametes.
344  ///
345  /// The final three parameters will be modified.
346  ///
347  /// \return A pair of mut_rec_intermediates, corresponding to what is passed on from
348  /// each parent.
349  ///
350  /// The operations are dispatched out to functions based on the population type. These
351  /// functions make calls to fwdpp::mutate_recombine.
352  ///
353  /// The parameter \a mutation_policy governs what types of mutations are entered
354  /// into the gametes of \a offspring. If the policy is fwdpp::ts::all_mutations, then
355  /// neutral and selected variants are placed in an offspring's gametes. If the policy is
356  /// fwdpp::ts::selected_variants_only, then only selected mutations are placed into the gametes.
357  /// Regardless of the policy, ALL mutations are contained in the return value, with the idea
358  /// that the caller will record them into a fwdpp::ts::table_collection.
359  ///
360  /// \note The function type genetic_param_holder::generate_mutations must return
361  /// std::vector<fwdpp::uint_t>, with the values reflecting the locations of new
362  /// mutations on \a pop.mutations. Further, this function must already bind
363  /// any relevant mutation rates (as this generate_offspring does not accept them
364  /// as arguments).
365  ///
366  /// \version 0.7.4 Added to fwdpp::ts.
367  ///
368  {
369  return detail::generate_offspring_details(
370  typename poptype::popmodel_t(), r, parents, mutation_policy,
371  pop, genetics, offspring);
372  }
