35  make_poisson_interlocus_rec(const gsl_rng* r, const double* means,
36  const std::size_t n)
37  /// \brief Create a vector of callbacks bound to model interlocus
38  /// recombination
39  ///
40  /// Example use for two loci with recombination occuring
41  /// at rate 1e-3 between them:
42  ///
43  /// \code
44  /// std::vector<double> recrates_bw_loci{1e-3};
45  /// auto interlocus_rec =
46  /// fwdpp::make_poisson_interlocus_rec(r,recrates_bw_loci.data(),recrates_bw_loci.size());
47  /// \endcode
48  ///
49  /// \ingroup mlocus
50  {
51  std::vector<std::function<unsigned(void)>> rv;
52  for (std::size_t i = 0; i < n; ++i)
53  {
54  auto mi = means[i];
55  rv.emplace_back([r, mi]() { return gsl_ran_poisson(r, mi); });
56  }
57  return rv;
58  }

63  /// \brief Create a vector of callbacks bound to model interlocus
64  /// recombination
65  ///
66  /// Example of a three locus system where loci 0 and 1
67  /// are 25cM apart and locus 2 is unlinked (50cM) from
68  /// 0 and 1:
69  ///
70  /// \code
71  /// std::vector<double> recrates{0.25,0.5};
72  /// auto interlocus_rec =
73  /// fwdpp::make_binomial_interlocus_rec(r,recrates.data(),recrates.size());
74  /// \endcode
75  ///
76  /// \ingroup mlocus
77  {
78  std::vector<std::function<unsigned(void)>> rv;
79  for (std::size_t i = 0; i < n; ++i)
80  {
81  auto di = distances[i];
82  rv.emplace_back(
83  [r, di]() { return gsl_ran_binomial(r, di, 1); });
84  }
85  return rv;
