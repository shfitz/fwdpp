57 void
58 expensive_leaf_test(const fwdpp::ts::table_collection &tables,
59  const std::vector<fwdpp::ts::TS_NODE_INT> &sample_list)
60 {
61  fwdpp::ts::tree_visitor mti(tables, sample_list);
62  while (mti(std::true_type(), std::true_type()))
63  {
64  auto &tree = mti.tree();
65  for (auto i : sample_list)
66  {
67  auto p = i;
68  while (p != -1)
69  {
70  auto l = tree.left_sample[p];
71  auto ogl = l;
72  if (l != -1)
73  {
74  auto r = tree.right_sample[p];
75  int ns = 0;
76  while (true)
77  {
78  ++ns;
79  if (l == r)
80  {
81  break;
82  }
83  l = tree.next_sample[l];
84  if (l == ogl)
85  {
86  throw std::runtime_error(
87  "loopback error");
88  }
89  }
90  if (ns != tree.leaf_counts[p])
91  {
92  throw std::runtime_error(
93  "bad sample interval");
94  }
95  }
96  p = tree.parents[p];
97  }
98  }
99  }

102 template <typename mcont_t>
103 void
104 matrix_runtime_test(const fwdpp::ts::table_collection &tables,
105  const std::vector<fwdpp::ts::TS_NODE_INT> &samples,
106  const mcont_t &mutations,
107  const std::vector<fwdpp::uint_t> &mcounts)
108 {
109  auto dm = fwdpp::ts::generate_data_matrix(tables, samples, mutations, true,
110  true);
111  auto rs = fwdpp::row_sums(dm);
112  for (std::size_t i = 0; i < rs.first.size(); ++i)
113  {
114  if (rs.first[i] != mcounts[dm.neutral_keys[i]])
115  {
116  throw std::runtime_error("bad neutral mutation count");
117  }
118  }
119  for (std::size_t i = 0; i < rs.second.size(); ++i)
120  {
121  if (rs.second[i] != mcounts[dm.selected_keys[i]])
122  {
123  throw std::runtime_error("bad selected mutation count");
124  }
125  }

167 template <typename rng>
168 std::function<double()>
169 make_dfe(const fwdpp::uint_t N, const rng &r, const double mean,
170  const double shape, const double scoeff)
171 {
172  if (std::isfinite(scoeff))
173  {
174  return [scoeff]() { return scoeff; };
175  }
176  fwdpp::extensions::gamma dfe(mean, shape);
177  return
178  [&r, dfe, N]() { return dfe(r.get()) / static_cast<double>(2 * N); };
179 }
