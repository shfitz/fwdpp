1 /* \include juvenile_migration.cc
2  *
3  * Juvenile migration.
4  * Selection.
5  * Constant pop size.
6  * Unequal migration rates.
7  *
8  * The goal is to show how to use a population
9  * object to handle details of population structure.
10  *
11  * The outline of the scheme is as follows:
12  *
13  * 1. Offspring are generated in sorted order by deme
14  * label. In other words, deme 1 before deme 2.
15  *
16  * 2. Each generation, we migrate first. Thus,
17  * we may start a generation with parents like this:
18  *
19  * Parent Deme
20  * 0 0
21  * 1 0
22  * 2 0
23  * 3 1
24  * 4 1
25  * 5 1
26  *
27  * After migration, we may end up with:
28  *
29  * Parent Deme
30  * 0 1*
31  * 1 0
32  * 2 0
33  * 3 1
34  * 4 0*
35  * 5 1
36  *
37  * An asterisk represent migrant individuals.
38  *
39  * We will generate efficient lookup tables to sample individuals
40  * proportional to their within-deme fitness, post-migration.
41  *
42  * These lookup tables will map to the indexes of the parents of each
43  * deme:
44  *
45  * Lookup 1:
46  *
47  * Index Parent
48  * 0 1
49  * 1 2
50  * 2 4
51  *
52  * Lookup 2:
53  * Index Parent
54  * 0 0
55  * 1 3
56  * 1 5
57  *
58  * The code is not safe for real-world use. It doesn't handle the corner
59  * case of all of deme 1 migrating into deme 2, leaving deme 1 "extinct"
60  * in the next generation. Such edge cases are clearly important, but
61  * the goal here it to show the book-keeping of parental fitnesses
62  * and the mapping back to deme labels.
63  */

81 struct parent_lookup_tables
82 // Our object for choosing parents each generation
83 {
84  // These return indexes of parents from demes 1 and 2,
85  // resp, chosen in O(1) time proportional to
86  // relative fitness within each deme
87  fwdpp_internal::gsl_ran_discrete_t_ptr lookup1, lookup2;

94 template <typename fitness_fxn>
95 parent_lookup_tables
96 migrate_and_calc_fitness(const gsl_rng *r, singlepop_t &pop,
97  const fitness_fxn &wfxn, const uint_t N1,
98  const uint_t N2, const double m12, const double m21)
99 // This function will be called at the start of each generation.
100 // The main goal is to return the lookup tables described above.
101 // But, "while we're at it", it does some other stuff that
102 // needs to be done at the start of each generation.
103 // Neither the most rigorous nor the most efficient:
104 // 1. Ignores probability of back-migration.
105 // 2. Allocates 4 vectors each generation.
106 {
107  parent_lookup_tables rv;
108 
109  // Temp containers for fitnesses in each deme,
110  // post-migration
111  std::vector<double> w1, w2;
112 
113  // Pick no. migrants 1 -> 2 and 2 -> 1.
114  unsigned nmig12 = gsl_ran_poisson(r, static_cast<double>(N1) * m12);
115  unsigned nmig21 = gsl_ran_poisson(r, static_cast<double>(N2) * m21);
116 
117  // Fill a vector of N1 zeros and N2 ones:
118  std::vector<uint_t> deme_labels(N1, 0);
119  deme_labels.resize(N1 + N2, 1);
120  assert(deme_labels.size() == pop.diploids.size());
121 
122  // Set up source and destination containers
123  // for sampling w/o replacement
124  std::vector<std::size_t> individuals(N1 + N2);
125  std::iota(std::begin(individuals), std::end(individuals), 0);
126  std::vector<std::size_t> migrants(std::max(nmig12, nmig21));
127 
128  // Who is migrating 1 -> 2?
129  gsl_ran_choose(r, migrants.data(), nmig12, individuals.data(), N1,
130  sizeof(std::size_t));
131  for (std::size_t i = 0; i < nmig12; ++i)
132  {
133  deme_labels[migrants[i]] = !deme_labels[migrants[i]];
134  }
135 
136  // Exact same logic for migrants 2 -> 1
137  gsl_ran_choose(r, migrants.data(), nmig21, individuals.data() + N1, N2,
138  sizeof(std::size_t));
139  for (std::size_t i = 0; i < nmig21; ++i)
140  {
141  deme_labels[migrants[i]] = !deme_labels[migrants[i]];
142  }
143 
144  // Go over all parents, set gametes counts to zero,
145  // and put individual IDs and fitnesses into
146  // the right vectors:
147  for (std::size_t i = 0; i < deme_labels.size(); ++i)
148  {
149  // fwdpp requires that we zero out gamete
150  // counts each generation. Since we're looping
151  // over diploids here, now is a good time to
152  // handle this task, which saves us from having to
153  // do another O(N1+N2) loop:
154  pop.gametes[pop.diploids[i].first].n
155  = pop.gametes[pop.diploids[i].second].n = 0;
156  if (deme_labels[i] == 0)
157  {
158  rv.parents1.push_back(i);
159  w1.push_back(
160  wfxn(pop.diploids[i], pop.gametes, pop.mutations));
161  }
162  else
163  {
164  rv.parents2.push_back(i);
165  w2.push_back(
166  wfxn(pop.diploids[i], pop.gametes, pop.mutations));
167  }
168  }
169 
170  // Set up our lookup tables:
171  rv.lookup1.reset(gsl_ran_discrete_preproc(rv.parents1.size(), w1.data()));
172  rv.lookup2.reset(gsl_ran_discrete_preproc(rv.parents2.size(), w2.data()));
173  return rv;
174 };

177 void
178 evolve_two_demes(const gsl_rng *r, singlepop_t &pop, const uint_t N1,
179  const uint_t N2, const double m12, const double m21,
180  const double mu, const fitness_fxn &wfxn,
181  const rec_fxn &recfxn, const mut_fxn &mutfxn)
182 {
183  // Handle mutation/gamete "recycling":
184  auto mut_recycling_bin = fwdpp_internal::make_mut_queue(pop.mcounts);
185  auto gam_recycling_bin = fwdpp_internal::make_gamete_queue(pop.gametes);
186 
187  // Migration and build lookup tables:
188  auto lookups = migrate_and_calc_fitness(r, pop, wfxn, N1, N2, m12, m21);
189 
190 #ifndef NDEBUG
191  for (const auto &g : pop.gametes)
192  assert(!g.n);
193 #endif
194 
195  // Copy parents
196  const auto parents(pop.diploids);
197 
198  // Fill in the next generation!
199  // We generate the offspring for deme 1 first,
200  // and then for deme 2
201  for (uint_t i = 0; i < N1 + N2; ++i)
202  {
203  std::size_t p1 = std::numeric_limits<std::size_t>::max();
204  std::size_t p2 = std::numeric_limits<std::size_t>::max();
205  if (i < N1) // pick parents from pop 1
206  {
207  p1 = lookups.parents1[gsl_ran_discrete(
208  r, lookups.lookup1.get())];
209  p2 = lookups.parents1[gsl_ran_discrete(
210  r, lookups.lookup1.get())];
211  }
212  else // pick parents from pop 2
213  {
214  p1 = lookups.parents2[gsl_ran_discrete(
215  r, lookups.lookup2.get())];
216  p2 = lookups.parents2[gsl_ran_discrete(
217  r, lookups.lookup2.get())];
218  }
219  assert(p1 < parents.size());
220  assert(p2 < parents.size());
221 
222  /*
223  These are the gametes from each parent.
224  */
225  auto p1g1 = parents[p1].first;
226  auto p1g2 = parents[p1].second;
227  auto p2g1 = parents[p2].first;
228  auto p2g2 = parents[p2].second;
229 
230  // "Mendel"
231  if (gsl_rng_uniform(r) < 0.5)
232  std::swap(p1g1, p1g2);
233  if (gsl_rng_uniform(r) < 0.5)
234  std::swap(p2g1, p2g2);
235 
236  mutate_recombine_update(r, pop.gametes, pop.mutations,
237  std::make_tuple(p1g1, p1g2, p2g1, p2g2),
238  recfxn, mutfxn, mu, gam_recycling_bin,
239  mut_recycling_bin, pop.diploids[i],
240  pop.neutral, pop.selected);
241  }
242  assert(check_sum(pop.gametes, 2 * (N1 + N2)));
243 #ifndef NDEBUG
244  for (const auto &dip : pop.diploids)
245  {
246  assert(pop.gametes[dip.first].n > 0);
247  assert(pop.gametes[dip.first].n <= 2 * (N1 + N2));
248  assert(pop.gametes[dip.second].n > 0);
249  assert(pop.gametes[dip.second].n <= 2 * (N1 + N2));
250  }
251 #endif
252 
253  // Update mutation counts
254  fwdpp_internal::process_gametes(pop.gametes, pop.mutations, pop.mcounts);
255 
256  assert(pop.mcounts.size() == pop.mutations.size());
257 #ifndef NDEBUG
258  for (const auto &mc : pop.mcounts)
259  {
260  assert(mc <= 2 * (N1 + N2));
261  }
262 #endif
263  assert(
264  popdata_sane(pop.diploids, pop.gametes, pop.mutations, pop.mcounts));
265 
266  // Prune fixations from gametes
267  fwdpp_internal::gamete_cleaner(pop.gametes, pop.mutations, pop.mcounts,
268  2 * (N1 + N2), std::true_type());
269 }
