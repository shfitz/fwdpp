183  serialize_tables(ostreamtype& o, const table_collection& tables)
184  /*! \brief Write a fwdpp::ts::table_collection to a binary format.
185  *
186  * \param ostreamtype A model of std::ostream
187  * \param tables A fwdpp::ts::table_collection
188  *
189  * The result is that the data in \a tables are written
190  * in a machine-readable format to \a o.
191  *
192  * This function always outputs to the format version
193  * specified by fwdpp::ts::io::TS_TABLES_VERSION.
194  */
195  {
196  o << "fwdppts";
197  fwdpp::io::scalar_writer sw;
198  sw(o, &TS_TABLES_VERSION);
199  auto L = tables.genome_length();
200  sw(o, &L);
201  sw(o, &tables.edge_offset);
202  std::size_t num_edges = tables.edge_table.size(),
203  num_nodes = tables.num_nodes(),
204  num_mutations = tables.mutation_table.size();
205  sw(o, &num_edges);
206  sw(o, &num_nodes);
207  sw(o, &num_mutations);
208  serialize_edge<TS_TABLES_VERSION> edge_writer;
209  serialize_node<TS_TABLES_VERSION> node_writer;
210  serialize_mutation_record<TS_TABLES_VERSION>
211  mutation_record_writer;
212  for (auto& e : tables.edge_table)
213  {
214  edge_writer(o, e);
215  }
216  for (auto& n : tables.node_table)
217  {
218  node_writer(o, n);
219  }
220  for (auto& m : tables.mutation_table)
221  {
222  mutation_record_writer(o, m);
223  }
224  std::size_t num_preserved_samples
225  = tables.preserved_nodes.size();
226  sw(o, &num_preserved_samples);
227  if (num_preserved_samples)
228  {
229  sw(o, tables.preserved_nodes.data(),

236  deserialize_tables(istreamtype& i)
