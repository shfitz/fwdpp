78  void
79  split_breakpoints(
80  const std::vector<double>& breakpoints,
81  const std::tuple<TS_NODE_INT, TS_NODE_INT>& parents,
82  const TS_NODE_INT next_index)
83  {
84  if (breakpoints.empty())
85  {
86  this->push_back_edge(0., L, std::get<0>(parents),
87  next_index);
88  return;
89  }
90  auto itr = std::adjacent_find(std::begin(breakpoints),
91  std::end(breakpoints));
92  if (itr == std::end(breakpoints))
93  {
94  split_breakpoints_add_edges(breakpoints, parents,
95  next_index);
96  }
97  else
98  {
99  // Here, we need to reduce the input
100  // breakpoints to only those seen
101  // an odd number of times.
102  // Even numbers of the same breakpoint
103  // are "double x-overs" and thus
104  // cannot affect the genealogy.
105  std::vector<double> odd_breakpoints;
106  auto start = breakpoints.begin();
107  while (itr < breakpoints.end())
108  {
109  auto not_equal
110  = std::find_if(itr, breakpoints.end(),
111  [itr](const double d) {
112  return d != *itr;
113  });
114  int even = (std::distance(itr, not_equal) % 2
115  == 0.0);
116  odd_breakpoints.insert(odd_breakpoints.end(),
117  start, itr + 1 - even);
118  start = not_equal;
119  itr = std::adjacent_find(
120  start, std::end(breakpoints));
121  }
122  odd_breakpoints.insert(odd_breakpoints.end(), start,
123  breakpoints.end());
124  split_breakpoints_add_edges(odd_breakpoints, parents,
125  next_index);
126  }
127  }

178  void
179  sort_edges() noexcept
180  /// Sort the edge table. On PARENT birth times.
181  /// The sorting differs from msprime here. The difference
182  /// is that we assume that birth times are recorded forward in
183  /// time rather than backwards.
184  ///
185  /// In between simplifications, edges are appended to the table
186  /// starting at edge_offset. edges prior to that point are,
187  /// by definition, sorted b/c they are the output of simplification.
188  /// Thus, we only sort the new edges and then perform a left rotate
189  /// w.r.to edge_offset to put the nodes in the correct final order.
190  /// The rotation is exactly edge_table.size() swaps.
191  {
192  std::sort(edge_table.begin() + edge_offset, edge_table.end(),
193  [this](const edge& a, const edge& b) {
194  auto ga = this->node_table[a.parent].time;
195  auto gb = this->node_table[b.parent].time;
196  if (ga == gb)
197  {
198  if (a.parent == b.parent)
199  {
200  if (a.child == b.child)
201  {
202  return a.left < b.left;
203  }
204  return a.child < b.child;
205  }
206  return a.parent < b.parent;
207  }
208  return ga > gb;
209  });
210  if (edge_offset > 0)
211  {
212  std::rotate(edge_table.begin(),
213  edge_table.begin() + edge_offset,
214  edge_table.end());
215  }
216  // TODO: allow for exceptions
217  // rather than assertions.
218  assert(edges_are_sorted());
219  }
